```sql
-- [1. 서브쿼리 버전 코드]

SELECT
  m.member_name,
  r.review_text,
  DATE_FORMAT(r.review_date, '%Y-%m-%d') AS review_date
FROM REST_REVIEW r
JOIN (
  -- 최다 리뷰 회원만 뽑아놓은 파생 테이블
  SELECT rc.member_id
  FROM (
    -- 회원별 리뷰 수
    SELECT member_id, COUNT(*) AS cnt
    FROM REST_REVIEW
    GROUP BY member_id
  ) rc
  JOIN (
    -- 최댓값
    SELECT MAX(cnt) AS mx
    FROM (
      SELECT member_id, COUNT(*) AS cnt
      FROM REST_REVIEW
      GROUP BY member_id
    ) x
  ) mm
  ON rc.cnt = mm.mx
) t  USING (member_id)
JOIN MEMBER_PROFILE m USING (member_id)
ORDER BY r.review_date ASC, r.review_text ASC;

```

```sql
-- [2. CTE 버전 코드]

-- 1) 회원별 리뷰수
WITH review_counts AS (
  SELECT member_id, COUNT(*) AS cnt
  FROM REST_REVIEW
  GROUP BY member_id
),
-- 2) 최다 리뷰 회원 id
top_members AS (
  SELECT member_id
  FROM review_counts
  WHERE cnt = (SELECT MAX(cnt) FROM review_counts)
)

SELECT
  m.member_name,
  r.review_text,
  DATE_FORMAT(r.review_date, '%Y-%m-%d') AS review_date
FROM REST_REVIEW r 
JOIN top_members t USING (member_id)
JOIN MEMBER_PROFILE m USING (member_id)
ORDER BY r.review_date ASC, r.review_text ASC;

```

```sql
-- [3. 윈도우함수(RANK) 버전 코드]

-- 회원별 리뷰 수 집계 후 순위 매기기
WITH review_counts AS (
  SELECT member_id, COUNT(*) AS cnt
  FROM REST_REVIEW
  GROUP BY member_id
),
ranked AS (
  SELECT member_id, cnt,
         RANK() OVER (ORDER BY cnt DESC) AS rnk
  FROM review_counts
)
SELECT
  m.member_name,
  r.review_text,
  DATE_FORMAT(r.review_date, '%Y-%m-%d') AS review_date
FROM ranked k
JOIN REST_REVIEW r USING (member_id)
JOIN MEMBER_PROFILE m USING (member_id)
WHERE k.rnk = 1
ORDER BY r.review_date ASC, r.review_text ASC;

```

### 비교 Point
1. **서브쿼리 버전 코드**
   - 가독성↓: 괄호 속 괄호… 중첩이 깊어지면 읽기/유지보수 난이도 상승.
2. **CTE 버전 코드**
   - 모듈화 → 가독성, 디버깅, 재사용, 유지보수 용이
3. **윈도우함수(RANK) 버전 코드**
   - 읽기 쉬운 표준 패턴이라 팀 협업에 유리.

### 코드 패턴 Point

- [비효율적] **조인 → 집계**
  - 모든 행 기준으로 조인된 뒤에야 집계가 일어나기 떄문에 불필요한 메모리 사용 유발
- [효율적] **집계(CTE) → 축소(필터링) → 조인**
  - 데이터를 축소시키고 조인하기 때문에 메모리 비용 절감
